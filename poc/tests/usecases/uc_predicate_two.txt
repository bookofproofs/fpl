        pred Equals(x,y: tpl)
        {
            p: pred a,b: tpl

            all a, b
            (
                iif
                (
                    Equals(a,b),
                    all p
                    (
                        iif
                        (
                            p(a),
                            p(b)
                        )
                    )
                )
            )
        }

        pred NotEquals(x,y: tpl)
        {
            not
            (
                Equals(x,y)
            )
        }

    }
}
